CONTENT struct TrieNode {
CONTENT     char ch;
CONTENT     vector<TrieNode*> children;
CONTENT     bool isEnd;
CONTENT     string word;
CONTENT 
CONTENT     TrieNode(char c = 0) : ch(c), isEnd(false), word("") {}
CONTENT };
CONTENT 
CONTENT TrieNode* trieRoot = nullptr;
CONTENT TrieNode* trieNewNode(char c) {
CONTENT     return new TrieNode(c);
CONTENT }
CONTENT 
CONTENT void trieInsert(const string& word) {
CONTENT     if (!trieRoot) trieRoot = trieNewNode(0);
CONTENT 
CONTENT     TrieNode* cur = trieRoot;
CONTENT     for (char c : word) {
CONTENT         TrieNode* nxt = nullptr;
CONTENT         for (auto child : cur->children) {
CONTENT             if (child->ch == c) {
CONTENT                 nxt = child;
CONTENT                 break;
CONTENT             }
CONTENT         }
CONTENT         if (!nxt) {
CONTENT             nxt = trieNewNode(c);
CONTENT             cur->children.push_back(nxt);
CONTENT         }
CONTENT         cur = nxt;
CONTENT     }
CONTENT     cur->isEnd = true;
CONTENT     cur->word = word;
CONTENT }
CONTENT 
CONTENT vector<string> trieSuggest(const string& prefix, int limit) {
CONTENT     vector<string> results;
CONTENT     if (!trieRoot || prefix.empty()) return results;
CONTENT 
CONTENT     TrieNode* cur = trieRoot;
CONTENT     for (char c : prefix) {
CONTENT         TrieNode* nxt = nullptr;
CONTENT         for (auto child : cur->children) {
CONTENT             if (child->ch == c) {
CONTENT                 nxt = child;
CONTENT                 break;
CONTENT             }
CONTENT         }
CONTENT         if (!nxt) return results;
CONTENT         cur = nxt;
CONTENT     }
CONTENT 
CONTENT     std::function<void(TrieNode*, string)> dfs = [&](TrieNode* node, string path) {
CONTENT         if ((int)results.size() >= limit) return;
CONTENT 
CONTENT         if (node->isEnd)
CONTENT             results.push_back(node->word);
CONTENT 
CONTENT         for (auto child : node->children)
CONTENT             dfs(child, path + child->ch);
CONTENT     };
CONTENT 
CONTENT     dfs(cur, prefix);
CONTENT     return results;
CONTENT }

CONTENT string joinPath(Node* node) {
CONTENT vector<string> parts;
CONTENT Node* cur = node;
CONTENT while (cur != NULL) {
CONTENT parts.insert(parts.begin(), cur->name);
CONTENT cur = cur->parent;
CONTENT }
CONTENT string out = "/";
CONTENT for (size_t i = 0; i < parts.size(); ++i) {
CONTENT out += parts[i];
CONTENT if (i + 1 < parts.size()) out += "/";

CONTENT }
CONTENT // find node by absolute path 
CONTENT Node* findNodeByPath(const string& path) {
CONTENT     if (path.empty()) return NULL;
CONTENT     // path components split by '/'
CONTENT     vector<string> parts;
CONTENT     string cur;
CONTENT     for (size_t i = 0; i < path.size(); ++i) {
CONTENT         char c = path[i];
CONTENT         if (c == '/') {
CONTENT             if (!cur.empty()) { parts.push_back(cur); cur.clear(); }
CONTENT         } else {
CONTENT             cur.push_back(c);
CONTENT         }
CONTENT     }
CONTENT     if (!cur.empty()) parts.push_back(cur);
CONTENT 
CONTENT     if (parts.empty()) return NULL;
CONTENT     // first must match ROOT name
CONTENT     if (parts[0] != ROOT->name) return NULL;
CONTENT 
CONTENT     Node* node = ROOT;
CONTENT     for (size_t i = 1; i < parts.size(); ++i) {
CONTENT         node = findChild(node, parts[i]);
CONTENT         if (!node) return NULL;
CONTENT     }
CONTENT     return node;
CONTENT }

CONTENT //-------------------------------->>>> For generating expression tree (infix,postfix,prefix)
CONTENT int prec(const string& op) {
CONTENT     if (op=="+"||op=="-") return 1;
CONTENT     if (op=="*"||op=="/") return 2;
CONTENT     if (op=="^") return 3;
CONTENT     return 0;
CONTENT }
CONTENT bool isOperator(const string& s) {
CONTENT     return s=="+"||s=="-"||s=="*"||s=="/"||s=="^";
CONTENT }
CONTENT 
CONTENT vector<string> tokenize(const string& expr) {
CONTENT     stringstream ss(expr);
CONTENT     vector<string> t;
CONTENT     string x;
CONTENT     while (ss >> x) t.push_back(x);
CONTENT     return t;
CONTENT }
CONTENT //for centring tree
CONTENT string stripANSI(const string& s) {
CONTENT     string res;
CONTENT     bool esc = false;
CONTENT 
CONTENT     for (size_t i = 0; i < s.size(); i++) {
CONTENT         if (s[i] == '\033')
CONTENT             esc = true;
CONTENT         else if (esc && s[i] == 'm')
CONTENT             esc = false;
CONTENT         else if (!esc)
CONTENT             res += s[i];
CONTENT     }
CONTENT     return res;
CONTENT }
CONTENT //postfix tree
CONTENT ExpNode* buildPostfixTree(const string& expr) {
CONTENT     stack<ExpNode*> st;
CONTENT     for (auto& t : tokenize(expr)) {
CONTENT         if (!isOperator(t))
CONTENT             st.push(new ExpNode(t));
CONTENT         else {
CONTENT             auto r = st.top(); st.pop();
CONTENT             auto l = st.top(); st.pop();
CONTENT             auto n = new ExpNode(t);
CONTENT             n->left = l;
CONTENT             n->right = r;
CONTENT             st.push(n);
CONTENT         }
CONTENT     }
CONTENT     return st.empty() ? nullptr : st.top();
CONTENT }
CONTENT //prefixtree
CONTENT ExpNode* buildPrefixTree(const string& expr) {
CONTENT     auto tokens = tokenize(expr);
CONTENT     stack<ExpNode*> st;
CONTENT 
CONTENT     for (int i = tokens.size() - 1; i >= 0; --i) {
CONTENT         if (!isOperator(tokens[i]))
CONTENT             st.push(new ExpNode(tokens[i]));
CONTENT         else {
CONTENT             auto l = st.top(); st.pop();
CONTENT             auto r = st.top(); st.pop();
CONTENT             auto n = new ExpNode(tokens[i]);
CONTENT             n->left = l;
CONTENT             n->right = r;
CONTENT             st.push(n);
CONTENT         }
CONTENT     }
CONTENT     return st.empty() ? nullptr : st.top();
CONTENT }
CONTENT //converting infix to postfix for infixtree
CONTENT string infixToPostfix(const string& expr) {
CONTENT     stack<string> st;
CONTENT     string out;
CONTENT 
CONTENT     for (auto& t : tokenize(expr)) {
CONTENT         if (!isOperator(t) && t!="(" && t!=")") out += t+" ";
CONTENT         else if (t=="(") st.push(t);
CONTENT         else if (t==")") {
CONTENT             while (!st.empty() && st.top()!="(") {
CONTENT                 out += st.top()+" "; st.pop();
CONTENT             }
CONTENT             st.pop();
CONTENT         }
CONTENT         else {
CONTENT             while (!st.empty() && prec(st.top())>=prec(t)) {
CONTENT                 out += st.top()+" "; st.pop();
CONTENT             }
CONTENT             st.push(t);
CONTENT         }
CONTENT     }
CONTENT     while (!st.empty()) {
CONTENT         out += st.top()+" "; st.pop();
CONTENT     }
CONTENT     return out;
CONTENT }
CONTENT //infixtree
CONTENT ExpNode* buildInfixTree(const string& expr) {
CONTENT     return buildPostfixTree(infixToPostfix(expr));
CONTENT }
CONTENT //create tree
CONTENT void genTree(
CONTENT     ExpNode* node,
CONTENT     string prefix,
CONTENT     bool isLast,
CONTENT     bool isRoot,
CONTENT     bool isLeft
CONTENT ) {
CONTENT     if (!node) return;
CONTENT 
CONTENT     string color =
CONTENT         isRoot ? CLR_ROOT :
CONTENT         isLeft ? CLR_LEFT :
CONTENT                  CLR_RIGHT;
CONTENT 
CONTENT     string line = prefix;
CONTENT 
CONTENT     if (!isRoot) {
CONTENT         line += string(CLR_LINE)
CONTENT               + (isLast ? "└── " : "├── ")
CONTENT               + CLR_RESET;
CONTENT     }
CONTENT 
CONTENT     line += color + node->val + CLR_RESET;
CONTENT     treeLines.push_back(line);
CONTENT 
CONTENT    string nextPrefix = prefix +
CONTENT     (isLast
CONTENT         ? "    "
CONTENT         : string(CLR_LINE) + "│" + CLR_RESET + "   ");
CONTENT 
CONTENT 
CONTENT     if (node->left)
CONTENT         genTree(node->left, nextPrefix, false, false, true);
CONTENT 
CONTENT     if (node->right)
CONTENT         genTree(node->right, nextPrefix, true, false, false);
CONTENT }

CONTENT //print created tree
CONTENT void printExpressionTree(ExpNode* root, const string& title) {
CONTENT     if (!root) return;
CONTENT 
CONTENT     treeLines.clear();
CONTENT     genTree(root, "", true, true, false);
CONTENT 
CONTENT     int maxLen = 0;
CONTENT     for (auto& l : treeLines)
CONTENT         maxLen = std::max(maxLen, (int)stripANSI(l).length());
CONTENT 
CONTENT     int width = getConsoleWidth();
CONTENT     int pad = std::max(0, (width - maxLen) / 2);
CONTENT 
CONTENT     cout << "\n";
CONTENT     cout << string((width - title.length()) / 2, ' ')
CONTENT          << CLR_ROOT << title << CLR_RESET << "\n\n";
CONTENT 
CONTENT     for (auto& l : treeLines)
CONTENT         cout << string(pad, ' ') << l << "\n";
CONTENT 
CONTENT     cout << "\n";
CONTENT }

CONTENT     }

CONTENT         // undoing add => remove bookmark by name stored in extra
CONTENT         string bname = a.extra;
CONTENT         Bookmark* b = bookmarksHead; Bookmark* prev = NULL;

CONTENT //------------------------------>>>>>>>>> Load workspace + bookmarks
CONTENT void loadWorkspace(Node* root, std::ifstream &in) {
CONTENT     // clear current children & bookmarks
CONTENT     root->children.clear();
CONTENT     currentNode = root;
CONTENT     while (bookmarksHead) { Bookmark* t = bookmarksHead; bookmarksHead = bookmarksHead->next; delete t; }
CONTENT 
CONTENT     vector<Node*> depthNodes;
CONTENT     depthNodes.resize(1);
CONTENT     depthNodes[0] = root;
CONTENT 
CONTENT     string line;
CONTENT     Node* lastFileNode = NULL;
CONTENT     bool inBookmarkSection = false;
CONTENT     while (std::getline(in, line)) {
CONTENT         if (line.empty()) continue;
CONTENT         if (!inBookmarkSection && line == "BOOKMARKS") { inBookmarkSection = true; continue; }
CONTENT         if (inBookmarkSection) {
CONTENT             if (line.rfind("BOOKMARK ", 0) == 0) {
CONTENT                 string rest = line.substr(9);
CONTENT                 size_t sep = rest.find(" | ");
CONTENT                 string bname, bpath;
CONTENT                 if (sep != string::npos) { bname = rest.substr(0, sep); bpath = rest.substr(sep + 3); }
CONTENT                 else {
CONTENT                     size_t sp = rest.find(' ');
CONTENT                     if (sp != string::npos) { bname = rest.substr(0, sp); bpath = rest.substr(sp + 1); }
CONTENT                     else { bname = rest; bpath = ""; }
CONTENT                 }
CONTENT                 if (!bpath.empty()) {
CONTENT                     Node* target = findNodeByPath(bpath);
CONTENT                     if (target) { Bookmark* nb = new Bookmark(bname, target); nb->next = bookmarksHead; bookmarksHead = nb; target->isBookmarked = true; }
CONTENT                 }
CONTENT             }
CONTENT             continue;
CONTENT         }
CONTENT         if (line.rfind("CONTENT ", 0) == 0) {
CONTENT             if (lastFileNode) lastFileNode->fileContent += line.substr(8) + "\n";
CONTENT             continue;
CONTENT         }
CONTENT         stringstream ss(line);
CONTENT         int depth;
CONTENT         char type;
CONTENT         int lockState;
CONTENT         if (!(ss >> depth >> type >> lockState)) continue;
CONTENT         string name;
CONTENT         getline(ss, name);
CONTENT         if (!name.empty() && name[0] == ' ') name = name.substr(1);
CONTENT         if (depth == 0) { root->name = name; root->isFolder = (type == 'D'); root->isLocked = (lockState == 1); lastFileNode = NULL; continue; }
CONTENT         if ((int)depth >= (int)depthNodes.size()) depthNodes.resize(depth + 1, NULL);
CONTENT         Node* parent = depthNodes[depth - 1];
CONTENT         if (!parent) { lastFileNode = NULL; continue; }
CONTENT         Node* newNode = new Node(name, (type == 'D'), parent);
CONTENT         newNode->isLocked = (lockState == 1);
CONTENT         parent->children.push_back(newNode);
CONTENT         depthNodes[depth] = newNode;
CONTENT         lastFileNode = (newNode->isFolder ? NULL : newNode);
CONTENT     }
CONTENT }

CONTENT void trieInsert(const string& word) {
CONTENT     if (!trieRoot) trieRoot = trieNewNode(0);
CONTENT 
CONTENT     TrieNode* cur = trieRoot;
CONTENT     for (char c : word) {
CONTENT         TrieNode* nxt = nullptr;

CONTENT                 nxt = child;
CONTENT                 break;
CONTENT             }

CONTENT             nxt = trieNewNode(c);
CONTENT             cur->children.push_back(nxt);
CONTENT         }
CONTENT         cur = nxt;
CONTENT     }
CONTENT     cur->isEnd = true;
CONTENT     cur->word = word;
CONTENT }

CONTENT vector<string> trieSuggest(const string& prefix, int limit) {
CONTENT     vector<string> results;
CONTENT     if (!trieRoot || prefix.empty()) return results;
CONTENT 
CONTENT     TrieNode* cur = trieRoot;
CONTENT     for (char c : prefix) {
CONTENT         TrieNode* nxt = nullptr;

CONTENT             if (child->ch == c) {
CONTENT                 nxt = child;
CONTENT                 break;
CONTENT             }
CONTENT         }
CONTENT         if (!nxt) return results;
CONTENT         cur = nxt;
CONTENT     }

CONTENT string joinPath(Node* node) {
CONTENT vector<string> parts;
CONTENT Node* cur = node;
CONTENT while (cur != NULL) {
CONTENT parts.insert(parts.begin(), cur->name);
CONTENT cur = cur->parent;
CONTENT }
CONTENT string out = "/";
CONTENT for (size_t i = 0; i < parts.size(); ++i) {
CONTENT out += parts[i];
CONTENT if (i + 1 < parts.size()) out += "/";
CONTENT }
CONTENT return out;
CONTENT }
CONTENT // find node by absolute path 
CONTENT Node* findNodeByPath(const string& path) {
CONTENT     if (path.empty()) return NULL;
CONTENT     // path components split by '/'
CONTENT     vector<string> parts;
CONTENT     string cur;
CONTENT     for (size_t i = 0; i < path.size(); ++i) {
CONTENT         char c = path[i];
CONTENT         if (c == '/') {
CONTENT             if (!cur.empty()) { parts.push_back(cur); cur.clear(); }
CONTENT         } else {
CONTENT             cur.push_back(c);
CONTENT         }
CONTENT     }
CONTENT     if (!cur.empty()) parts.push_back(cur);
CONTENT 
CONTENT     if (parts.empty()) return NULL;
CONTENT     // first must match ROOT name
CONTENT     if (parts[0] != ROOT->name) return NULL;
CONTENT 
CONTENT     Node* node = ROOT;
CONTENT     for (size_t i = 1; i < parts.size(); ++i) {
CONTENT         node = findChild(node, parts[i]);
CONTENT         if (!node) return NULL;
CONTENT     }
CONTENT     return node;
CONTENT }

CONTENT //-------------------------------->>>> For generating expression tree (infix,postfix,prefix)
CONTENT int prec(const string& op) {
CONTENT     if (op=="+"||op=="-") return 1;
CONTENT     if (op=="*"||op=="/") return 2;
CONTENT     if (op=="^") return 3;
CONTENT     return 0;
CONTENT }
CONTENT bool isOperator(const string& s) {
CONTENT     return s=="+"||s=="-"||s=="*"||s=="/"||s=="^";
CONTENT }
CONTENT 
CONTENT vector<string> tokenize(const string& expr) {
CONTENT     stringstream ss(expr);
CONTENT     vector<string> t;
CONTENT     string x;
CONTENT     while (ss >> x) t.push_back(x);
CONTENT     return t;
CONTENT }
CONTENT //for centring tree
CONTENT string stripANSI(const string& s) {
CONTENT     string res;
CONTENT     bool esc = false;
CONTENT 
CONTENT     for (size_t i = 0; i < s.size(); i++) {
CONTENT         if (s[i] == '\033')
CONTENT             esc = true;
CONTENT         else if (esc && s[i] == 'm')
CONTENT             esc = false;
CONTENT         else if (!esc)
CONTENT             res += s[i];
CONTENT     }
CONTENT     return res;
CONTENT }
CONTENT //postfix tree
CONTENT ExpNode* buildPostfixTree(const string& expr) {
CONTENT     stack<ExpNode*> st;
CONTENT     for (auto& t : tokenize(expr)) {
CONTENT         if (!isOperator(t))
CONTENT             st.push(new ExpNode(t));
CONTENT         else {
CONTENT             auto r = st.top(); st.pop();
CONTENT             auto l = st.top(); st.pop();
CONTENT             auto n = new ExpNode(t);
CONTENT             n->left = l;
CONTENT             n->right = r;
CONTENT             st.push(n);
CONTENT         }
CONTENT     }
CONTENT     return st.empty() ? nullptr : st.top();
CONTENT }
CONTENT //prefixtree
CONTENT ExpNode* buildPrefixTree(const string& expr) {
CONTENT     auto tokens = tokenize(expr);
CONTENT     stack<ExpNode*> st;
CONTENT 
CONTENT     for (int i = tokens.size() - 1; i >= 0; --i) {
CONTENT         if (!isOperator(tokens[i]))
CONTENT             st.push(new ExpNode(tokens[i]));
CONTENT         else {
CONTENT             auto l = st.top(); st.pop();
CONTENT             auto r = st.top(); st.pop();
CONTENT             auto n = new ExpNode(tokens[i]);
CONTENT             n->left = l;
CONTENT             n->right = r;
CONTENT             st.push(n);
CONTENT         }
CONTENT     }
CONTENT     return st.empty() ? nullptr : st.top();
CONTENT }
CONTENT //converting infix to postfix for infixtree
CONTENT string infixToPostfix(const string& expr) {
CONTENT     stack<string> st;
CONTENT     string out;
CONTENT 
CONTENT     for (auto& t : tokenize(expr)) {
CONTENT         if (!isOperator(t) && t!="(" && t!=")") out += t+" ";
CONTENT         else if (t=="(") st.push(t);
CONTENT         else if (t==")") {
CONTENT             while (!st.empty() && st.top()!="(") {
CONTENT                 out += st.top()+" "; st.pop();
CONTENT             }
CONTENT             st.pop();
CONTENT         }
CONTENT         else {
CONTENT             while (!st.empty() && prec(st.top())>=prec(t)) {
CONTENT                 out += st.top()+" "; st.pop();
CONTENT             }
CONTENT             st.push(t);
CONTENT         }
CONTENT     }
CONTENT     while (!st.empty()) {
CONTENT         out += st.top()+" "; st.pop();
CONTENT     }
CONTENT     return out;
CONTENT }
CONTENT //infixtree
CONTENT ExpNode* buildInfixTree(const string& expr) {
CONTENT     return buildPostfixTree(infixToPostfix(expr));
CONTENT }
CONTENT //create tree
CONTENT void genTree(
CONTENT     ExpNode* node,
CONTENT     string prefix,
CONTENT     bool isLast,
CONTENT     bool isRoot,
CONTENT     bool isLeft
CONTENT ) {
CONTENT     if (!node) return;
CONTENT 
CONTENT     string color =
CONTENT         isRoot ? CLR_ROOT :
CONTENT         isLeft ? CLR_LEFT :
CONTENT                  CLR_RIGHT;
CONTENT 
CONTENT     string line = prefix;
CONTENT 
CONTENT     if (!isRoot) {
CONTENT         line += string(CLR_LINE)
CONTENT               + (isLast ? "└── " : "├── ")
CONTENT               + CLR_RESET;
CONTENT     }
CONTENT 
CONTENT     line += color + node->val + CLR_RESET;
CONTENT     treeLines.push_back(line);
CONTENT 
CONTENT    string nextPrefix = prefix +
CONTENT     (isLast
CONTENT         ? "    "
CONTENT         : string(CLR_LINE) + "│" + CLR_RESET + "   ");
CONTENT 
CONTENT 
CONTENT     if (node->left)
CONTENT         genTree(node->left, nextPrefix, false, false, true);
CONTENT 
CONTENT     if (node->right)
CONTENT         genTree(node->right, nextPrefix, true, false, false);
CONTENT }

CONTENT void searchDFS(Node* node, const string& keyword, vector<string>& results, const string& path="") {
CONTENT if (!node) return;
CONTENT string currentPath = path + "/" + node->name;

CONTENT void searchBFS(Node* start, const string& keyword, vector<string>& results) {
CONTENT if (!start) return;
CONTENT vector<Node*> queue;
CONTENT vector<string> paths;
CONTENT queue.push_back(start);
CONTENT paths.push_back("/" + start->name);
CONTENT 
CONTENT 
CONTENT for (size_t i = 0; i < queue.size(); ++i) {
CONTENT     Node* node = queue[i];
CONTENT     string path = paths[i];
CONTENT     if (node->name.find(keyword) != string::npos) results.push_back(path);
CONTENT     for (size_t j = 0; j < node->children.size(); ++j) {
CONTENT         queue.push_back(node->children[j]);
CONTENT         paths.push_back(path + "/" + node->children[j]->name);
CONTENT     }
CONTENT   }
CONTENT }

CONTENT     if (undoStack.empty()) { center("⚠ Nothing to undo"); return; }
CONTENT     Action a = undoStack.top(); undoStack.pop();
CONTENT 
CONTENT     if (a.type == "create") {
CONTENT         if (a.parent) {
CONTENT             auto &children = a.parent->children;
CONTENT             auto it = std::find(children.begin(), children.end(), a.node);
CONTENT             if (it != children.end()) children.erase(it);
CONTENT             
CONTENT             if (a.node && !a.node->isFolder) deletePhysicalFile(a.node->name);
CONTENT             redoStack.push(Action("create", a.node, a.parent, a.index, ""));
CONTENT             center("↩️ Undo: removed created item");
CONTENT         } else center("⚠ Undo error: parent missing");
CONTENT     }
CONTENT     else if (a.type == "delete") {
CONTENT         if (a.parent) {
CONTENT             auto &children = a.parent->children;
CONTENT             int idx = a.index;
CONTENT             if (idx < 0 || idx > (int)children.size()) idx = (int)children.size();
CONTENT             children.insert(children.begin() + idx, a.node);
CONTENT             a.node->parent = a.parent;
CONTENT             if (!a.node->isFolder) createPhysicalFile(a.node->name);
CONTENT             redoStack.push(Action("delete", a.node, a.parent, idx, ""));
CONTENT             center("↩️ Undo: restored deleted item");
CONTENT         } else center("⚠ Undo error: parent missing");
CONTENT     }
CONTENT     else if (a.type == "bookmark_add") {
CONTENT         string bname = a.extra;
CONTENT         Bookmark* b = bookmarksHead; Bookmark* prev = NULL;
CONTENT         while (b) {
CONTENT             if (b->name == bname) {
CONTENT                 if (b->node) b->node->isBookmarked = false;
CONTENT                 if (prev) prev->next = b->next; else bookmarksHead = b->next;
CONTENT                 delete b;
CONTENT                 redoStack.push(Action("bookmark_add", nullptr, nullptr, -1, bname));
CONTENT                 center("↩️ Undo: removed bookmark " + bname);
CONTENT                 return;
CONTENT             }
CONTENT             prev = b; b = b->next;
CONTENT         }
CONTENT         center("⚠ Undo bookmark_add: not found");
CONTENT     }
CONTENT     else if (a.type == "bookmark_remove") {
CONTENT     	
CONTENT         string bname = a.extra;
CONTENT         std::function<Node*(Node*)> findByName = [&](Node* nd)->Node* {
CONTENT             if (!nd) return nullptr;
CONTENT             if (nd->name == bname) return nd;
CONTENT             for (auto ch : nd->children) {
CONTENT                 Node* r = findByName(ch);
CONTENT                 if (r) return r;
CONTENT             }
CONTENT             return nullptr;
CONTENT         };
CONTENT         Node* target = findByName(ROOT);
CONTENT         if (target) {
CONTENT             Bookmark* nb = new Bookmark(bname, target);
CONTENT             nb->next = bookmarksHead; bookmarksHead = nb;
CONTENT             target->isBookmarked = true;
CONTENT             redoStack.push(Action("bookmark_remove", nullptr, nullptr, -1, bname));
CONTENT             center("↩️ Undo: restored bookmark " + bname);
CONTENT         } else center("⚠ Undo bookmark_remove: original target not found");
CONTENT     }
CONTENT     else if (a.type == "rename") {
CONTENT         string oldNew = a.extra;
CONTENT         size_t sep = oldNew.find('|');
CONTENT         string oldName = (sep==string::npos? oldNew : oldNew.substr(0, sep));
CONTENT         string newName = (sep==string::npos? string() : oldNew.substr(sep+1));
CONTENT         if (a.node) {
CONTENT             a.node->name = oldName;
CONTENT             redoStack.push(Action("rename", a.node, a.parent, a.index, oldName + "|" + newName));
CONTENT             center("↩️ Undo: rename reverted to " + oldName);
CONTENT         } else center("⚠ Undo rename: node missing");
CONTENT     }
CONTENT     else {
CONTENT         center("⚠ Unsupported undo action");
CONTENT     }

CONTENT //------------------------------>>>>>>>>> Load workspace + bookmarks
CONTENT void loadWorkspace(Node* root, std::ifstream &in) {
CONTENT     root->children.clear();
CONTENT     currentNode = root;
CONTENT     while (bookmarksHead) {
CONTENT 	 Bookmark* t = bookmarksHead;
CONTENT 	  bookmarksHead = bookmarksHead->next;
CONTENT 	  delete t; 
CONTENT 	  }
CONTENT 
CONTENT     vector<Node*> depthNodes;
CONTENT     depthNodes.resize(1);
CONTENT     depthNodes[0] = root;
CONTENT 
CONTENT     string line;
CONTENT     Node* lastFileNode = NULL;
CONTENT     bool inBookmarkSection = false;
CONTENT     while (std::getline(in, line)) {
CONTENT         if (line.empty()) continue;
CONTENT         if (!inBookmarkSection && line == "BOOKMARKS") {
CONTENT 		 inBookmarkSection = true; 
CONTENT 		 continue;
CONTENT 		  }
CONTENT         if (inBookmarkSection) {
CONTENT             if (line.rfind("BOOKMARK ", 0) == 0) {
CONTENT                 string rest = line.substr(9);
CONTENT                 size_t sep = rest.find(" | ");
CONTENT                 string bname, bpath;
CONTENT                 if (sep != string::npos) { 
CONTENT 				bname = rest.substr(0, sep);
CONTENT 				 bpath = rest.substr(sep + 3);
CONTENT 				  }
CONTENT                 else {
CONTENT                     size_t sp = rest.find(' ');
CONTENT                     if (sp != string::npos) {
CONTENT 					 bname = rest.substr(0, sp); 
CONTENT 					 bpath = rest.substr(sp + 1);
CONTENT 					  }
CONTENT                     else { 
CONTENT 					bname = rest; 
CONTENT 					bpath = "";
CONTENT 					 }
CONTENT                 }
CONTENT                 if (!bpath.empty()) {
CONTENT                     Node* target = findNodeByPath(bpath);
CONTENT                     if (target) { 
CONTENT 					Bookmark* nb = new Bookmark(bname, target); 
CONTENT 					nb->next = bookmarksHead; bookmarksHead = nb; 
CONTENT 					target->isBookmarked = true; 
CONTENT 					}
CONTENT                 }
CONTENT             }
CONTENT             continue;
CONTENT         }
CONTENT         if (line.rfind("CONTENT ", 0) == 0) {
CONTENT             if (lastFileNode) lastFileNode->fileContent += line.substr(8) + "\n";
CONTENT             continue;
CONTENT         }
CONTENT         stringstream ss(line);
CONTENT         int depth;
CONTENT         char type;
CONTENT         int lockState;
CONTENT         if (!(ss >> depth >> type >> lockState)) continue;
CONTENT         string name;
CONTENT         getline(ss, name);
CONTENT         if (!name.empty() && name[0] == ' ') name = name.substr(1);
CONTENT         if (depth == 0) { 
CONTENT 		root->name = name;
CONTENT 		root->isFolder = (type == 'D'); 
CONTENT 		root->isLocked = (lockState == 1); 
CONTENT 		lastFileNode = NULL;
CONTENT 		continue; 
CONTENT 		  }
CONTENT         if ((int)depth >= (int)depthNodes.size()) depthNodes.resize(depth + 1, NULL);
CONTENT         Node* parent = depthNodes[depth - 1];
CONTENT         if (!parent) { 
CONTENT 		lastFileNode = NULL; 
CONTENT 		continue; 
CONTENT 		}
CONTENT         Node* newNode = new Node(name, (type == 'D'), parent);
CONTENT         newNode->isLocked = (lockState == 1);
CONTENT         parent->children.push_back(newNode);
CONTENT         depthNodes[depth] = newNode;
CONTENT         lastFileNode = (newNode->isFolder ? NULL : newNode);
CONTENT     }
CONTENT }
